const Handlebars = require('handlebars');
const fs = require('fs');

Handlebars.registerHelper('gt', function (a, b) {
    return a > b;
});

const test1 = JSON.parse(fs.readFileSync('test1.json', 'utf8'));
const test2 = JSON.parse(fs.readFileSync('test2.json', 'utf8'));

const template = fs.readFileSync('template.hbs', 'utf8');
const compiledTemplate = Handlebars.compile(template);
const output = compiledTemplate({ test1, test2 });

console.log(output);
